二叉树的遍历

```c++
typedef struct BiTNode{
    ElemType data;
    struct BiTNode *lchild,*rchild;
}BiTNode,*BiTree;

//先序遍历（递归）
void PreOrder(BiTree T){
    if(T!=NULL){
        visit(T);//根
        PreOrder(T->lchild);//左
        PreOrder(T->lchild);//右
    }
}
```
