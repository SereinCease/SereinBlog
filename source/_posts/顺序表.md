有序表插入仍然保持有序(升序）

```c
int List_insert(SqList &L,ElemType e){
    int i;
    for ( i = 0; i < L.length; i++) {
        if(e<L.data[i])
            break;
    }
    for (int j =L.length; j>=i+1 ; j--) {
        L.data[j]=L.data[j-1];
    }
    L.data[i]=e;
    L.length++;
    return i+1;
}
```

删除i位置的元素

```
int ListDelete(SqList &L,int i){
    int e;
    e=L.data[i-1];
    for (int j = i; j < L.length; j++) {
        L.data[j-1]=L.data[j];
    }
    L.length--;
    return e;
}
```
