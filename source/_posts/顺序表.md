# 顺序表的定义

```
#define MaxSize 50
typedef int ElemType;
typedef struct{
	ElemType data[MaxSize];
	int length;
}SqList;
```

有序表插入仍然保持有序(升序）

```c
int List_insert(SqList &L,ElemType e){
    int i;
    for ( i = 0; i < L.length; i++) {
        if(e<L.data[i])
            break;
    }
    for (int j =L.length; j>=i+1 ; j--) {
        L.data[j]=L.data[j-1];
    }
    L.data[i]=e;
    L.length++;
    return i+1;
}
```

顺序表的插入

```
bool ListInsert(SqList &L,int i,ElemType element){
    //判断插入位置是否合法
    if(i<1||i>L.length+1)
        return false;
    //判断顺序表是否满了
    if(L.length>=MaxSize)
        return false;
    //顺序表插入操作
    for (int j = L.length; j >=i; j--) {
        L.data[j]=L.data[j-1];
    }
    L.data[i-1]=element;
    L.length++;
    return true;
}
```

删除i位置的元素

```
int ListDelete(SqList &L,int i){
    int e;
    e=L.data[i-1];
    for (int j = i; j < L.length; j++) {
        L.data[j-1]=L.data[j];
    }
    L.length--;
    return e;
}
```
